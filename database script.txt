drop database if exists shoesarcar;

create database shoesarcar;
use shoesarcar;

CREATE TABLE IF NOT EXISTS carinfo (
	carplatenumber VARCHAR(10) NOT NULL,
	brand VARCHAR(50) NOT NULL,
	model VARCHAR(50) NOT NULL,
	yearofmanufactured DATE NOT NULL,
	milleage INT NOT NULL,
	price INT NOT NULL,
	transmission VARCHAR(8) NOT NULL,
	dealercode INT NOT NULL,
	enginecapacity VARCHAR(8) NOT NULL,
	PRIMARY KEY(carplatenumber)
);
